カウントダウン終了まで

html

<div id="timeLeft"></div>

js

<script>
(function(){
	function countDown() {
		var startDateTime = new Date();//現在時刻
		var endDateTime = new Date("may 01,2015 00:00:00");//終了時刻
		var leftTime = endDateTime - startDateTime;//残時間
		var oneday_mirisecond = 24 * 60 * 60 * 1000;//1日のミリ秒数

		// 残日数
		var cdDay = Math.floor(leftTime / oneday_mirisecond);

		// 残時間数
		var cdHour = Math.floor((leftTime % oneday_mirisecond) / (60 * 60 * 1000));

		// 残分数
		var cdMinute = Math.floor((leftTime % oneday_mirisecond) / (60 * 1000)) % 60;

		// 残秒数
		var cdSecond = Math.floor((leftTime % oneday_mirisecond) / 1000) % 60 % 60;

		$("#timeLeft").text(cdDay + '日' + cdHour + '時間' + cdMinute + '分' + cdSecond + '秒');
	}
	$(function() {
		countDown();
		setInterval(function(){
			countDown();
		}, 1000);
	});
}());
</script>
