※PC→SP該当ページへ

(function($){

	//userAgent
	var ua = navigator.userAgent;
	var smartFlg = (ua.indexOf('iPhone') > 0 && ua.indexOf('iPad') == -1) || ua.indexOf('iPod') > 0 || (ua.indexOf('Android') > 0 && ua.indexOf('Mobile') > 0 || ua.indexOf('Windows Phone') > 0);

	//url
	var url = document.location.href.split('.jp/').pop();
	var topFlg = (url==''||url=='index.html');
	var hasSpPageFlg = (url.indexOf('topic') === -1 && url.indexOf('inquiry') === -1);

	//pcViewCookie
	var pcViewCookie = $.cookie('xx_pc');

	//sp top reload
	if(topFlg && smartFlg && (pcViewCookie!='true')){
		window.location.href = './sp/index.html';
	}
	//sp page reload
	if(hasSpPageFlg && !topFlg && smartFlg && (pcViewCookie!='true')){
		location.href = '/sp/'+ url;
	}

	$(function(){

		//sp bnr add
		if (smartFlg && hasSpPageFlg) {
			$('#container').prepend('<div id="smpBnr"><a href="#">スマートフォンサイトへ</a></div>');
			$('#smpBnr a').on('click',function(e){
				e.preventDefault();
				$.cookie('xx_pc', false, {path:'/', expires:365});
				pcViewCookie = $.cookie('xx_pc');
				location.href = './sp/' + url;
			});
		}

	});


})(jQuery);

		e.preventDefault();

		//pc cookie make

		pcViewCookie = $.cookie('xx_pc');

	});

});
