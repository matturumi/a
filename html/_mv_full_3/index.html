js

<script>
$(function(){

		//keyVisual
		var keyV = $("#keyVisual"),
				slideSize = keyV.find(".slide").size();
		if( slideSize == 2 || slideSize == 3){
			//スライドが2つか3つの時
			var clone = keyV.children("div").clone();
			keyV.prepend(clone);
		}
		//スライドが1つじゃない時に実行
		if( slideSize != 1){
			keyV.carouFredSel({
				width: '100%',
				align: 'center',
				scroll		: {
					items		: 1,
					fx			: "linear",
					duration: 400
				},
				items: {
						visible: 3,
						start: -1
				},
				swipe : {
					onTouch : false
				},
				auto		: {
					timeoutDuration: 7000
				},
				prev	: {	
					button	: "#keyV_prev",
					key		: "left"
				},
				next	: { 
					button	: "#keyV_next",
					key		: "right"
				},
				onCreate : function(){
					//1枚目を中央にしてから表示
					var preppendLast = $(this).children("div:last");
					$(this).prepend(preppendLast);
					$("#keyVisualWrap").append('<div id="mvHide"><div class="hideLeft"></div><div class="hideRight"></div></div>');
				}
			});
		}

});
</script>
