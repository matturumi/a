youtube

動画

$(".youtube").colorbox({
	iframe:true,
	innerWidth:970,
	innerHeight:600,
	onOpen: function(){
		$("#cboxClose").css("opacity",0);
	},
	onComplete: function(){
		setTimeout(function(){
			$("#cboxClose").css("opacity",1);
		},2000);
	}
});

コールバック onOpen、onLoad、onComplete、onCleanup、onClosed

ボタン

$(".colorbox3").colorbox({
	onOpen:function(){
		alert('onOpen: 開こうとしてる');
	},
	onLoad:function(){
		alert('onLoad: コンテンツロードの開始');
	},
	onComplete:function(){
		alert('onComplete: 表示完了');
	},
	onCleanup:function(){
		alert('onCleanup: クローズ処理を開始');
	},
	onClosed:function(){
		alert('onClosed: クローズ完了');
	}
});
